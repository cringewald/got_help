########## 6. Join max to pivot and calculate avg and id using an array ###############

county_wide <- full_join(county_wide, max) %>%
  
  mutate(
    aian_diff = abs(max - aian_rate), 
    black_diff = abs(max - black_rate), 
    asian_diff = abs(max - asian_rate), 
    pacisl_diff = abs(max - pacisl_rate), 
    other_diff= abs(max - other_rate),
    twoormor_diff= abs(max - twoormor_rate), 
    nh_white_diff = abs(max - nh_white_rate), 
    latino_diff = abs(max - latino_rate)
    )
## Joining, by = c("GEOID", "NAME")
x <- array(c(county_wide$aian_diff, county_wide$black_diff, county_wide$asian_diff, county_wide$pacisl_diff, 
           county_wide$other_diff, county_wide$twoormor_diff, county_wide$nh_white_diff, county_wide$latino_diff), dim = c(58,8,1))
  
county_wide <- county_wide %>%

mutate(
  avg = rowSums(x, na.rm = TRUE) / county_wide$values_count,
  index_of_disparity = rowSums(x, na.rm = TRUE) / max / (county_wide$values_count - 1) * 100
)

This section calculates variance

########## 7. Calculate variance using another array and z-scores, quadrants, and ranks ###############
  
y <- (x - county_wide$avg) ^ 2

county_wide <- county_wide %>%
  
  mutate(
    variance = rowSums(y, na.rm = TRUE) / county_wide$values_count,
    aian_disparity_z = (county_wide$aian_diff - avg) / sqrt(variance),
    black_disparity_z = (county_wide$black_diff - avg) / sqrt(variance),
    asian_disparity_z = (county_wide$asian_diff - avg) / sqrt(variance),
    pacisl_disparity_z = (county_wide$pacisl_diff - avg) / sqrt(variance),
    other_disparity_z = (county_wide$other_diff - avg) / sqrt(variance),
    twoormor_disparity_z = (county_wide$twoormor_diff - avg) / sqrt(variance),
    nh_white_disparity_z = (county_wide$nh_white_diff - avg) / sqrt(variance),
    latino_disparity_z = (county_wide$latino_diff - avg) / sqrt(variance),
    
    aian_performance_z = (county_wide$aian_rate - mean(county_wide$aian_rate, na.rm = TRUE))  / sd(county_wide$aian_rate, na.rm = TRUE),
    black_performance_z = (county_wide$black_rate - mean(county_wide$black_rate, na.rm = TRUE))  / sd(county_wide$black_rate, na.rm = TRUE),
    asian_performance_z = (county_wide$asian_rate - mean(county_wide$asian_rate, na.rm = TRUE))  / sd(county_wide$asian_rate, na.rm = TRUE),
    pacisl_performance_z = (county_wide$pacisl_rate - mean(county_wide$pacisl_rate, na.rm = TRUE))  / sd(county_wide$pacisl_rate, na.rm = TRUE),
    other_performance_z = (county_wide$other_rate - mean(county_wide$other_rate, na.rm = TRUE))  / sd(county_wide$other_rate, na.rm = TRUE),
    twoomor_performance_z = (county_wide$twoormor_rate - mean(county_wide$twoormor_rate, na.rm = TRUE))  / sd(county_wide$twoormor_rate, na.rm = TRUE),
    nh_white_performance_z = (county_wide$nh_white_rate - mean(county_wide$nh_white_rate, na.rm = TRUE))  / sd(county_wide$nh_white_rate, na.rm = TRUE),
    latino_performance_z = (county_wide$latino_rate - mean(county_wide$latino_rate, na.rm = TRUE))  / sd(county_wide$latino_rate, na.rm = TRUE),
  
    performance_z = (county_wide$total_rate - mean(county_wide$total_rate, na.rm = TRUE))  / sd(county_wide$total_rate, na.rm = TRUE),
    disparity_z = (county_wide$index_of_disparity - mean(county_wide$index_of_disparity, na.rm = TRUE))  / sd(county_wide$index_of_disparity, na.rm = TRUE),
  )

county_wide <- county_wide %>%
  
  mutate(
    quadrant = case_when(
      county_wide$performance_z < 0 & county_wide$disparity_z <= 0 ~ "yellow",
      county_wide$performance_z < 0 & county_wide$disparity_z > 0 ~ "red",
      county_wide$performance_z >= 0 & county_wide$disparity_z > 0 ~ "orange",
      county_wide$performance_z >= 0 & county_wide$disparity_z <= 0 ~ "green"),
    
    performance_rank = rank(-county_wide$performance_z, na.last = TRUE, ties.method = "max"),
    disparity_rank = rank(-county_wide$index_of_disparity, na.last = TRUE, ties.method = "max")
    )
